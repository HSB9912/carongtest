-- ============================================ -- 관리자 삭제 권한 RLS 업데이트 -- 기존 정책 삭제 후 재생성 -- Supabase SQL Editor에서 실행 -- ============================================ -- 게시글: 본인 또는 관리자(admin_whitelist에 있는 이메일) 삭제 가능 DROP POLICY IF EXISTS "board_posts_delete" ON board_posts; CREATE POLICY "board_posts_delete" ON board_posts FOR DELETE USING ( auth.email() = author_email OR auth.email() IN (SELECT email FROM admin_whitelist WHERE role = 'admin') ); -- 게시글 수정도 관리자 가능 DROP POLICY IF EXISTS "board_posts_update" ON board_posts; CREATE POLICY "board_posts_update" ON board_posts FOR UPDATE USING ( auth.email() = author_email OR auth.email() IN (SELECT email FROM admin_whitelist WHERE role = 'admin') ); -- 댓글: 본인 또는 관리자 삭제 DROP POLICY IF EXISTS "board_comments_delete" ON board_comments; CREATE POLICY "board_comments_delete" ON board_comments FOR DELETE USING ( auth.email() = author_email OR auth.email() IN (SELECT email FROM admin_whitelist WHERE role = 'admin') ); -- 추천: 본인 또는 관리자 삭제 DROP POLICY IF EXISTS "board_likes_delete" ON board_likes; CREATE POLICY "board_likes_delete" ON board_likes FOR DELETE USING ( auth.email() = user_email OR auth.email() IN (SELECT email FROM admin_whitelist WHERE role = 'admin') ); -- 버디팀: 관리자만 삭제 (CASCADE로 미션/룰렛도 같이 삭제됨) DROP POLICY IF EXISTS "buddy_teams_delete" ON buddy_teams; CREATE POLICY "buddy_teams_delete" ON buddy_teams FOR DELETE USING ( auth.email() IN (SELECT email FROM admin_whitelist WHERE role = 'admin') );
